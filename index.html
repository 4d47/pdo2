The 5k non-mapping database framework toolkit for rapid development.

$db = new PDO2(new PDO($dsn));
$db->select('foo');
$db->select('id, name FROM foo');
$db->select('foo', array('first_name' => 'Al'));
$db->insert('foo', $data);
$db->update('foo', $data, array('first_name' => 'Al'));
$db->delete('foo', array('first_name' => 'Al'));
$db->count('foo', array('first_name' => 'Al'));
